SETUP — Google Sheets 연동

연동이 필요한 이유
⚠️ 연동하지 않으면 인증 기록이 브라우저를 닫는 순간 사라집니다.
✅ 연동 시 인증 완료마다 Google Sheets에 자동 저장 + 중복 인증 서버 측 차단이 가능합니다.
연동 방법 (최초 1회)
1
Google Sheets 생성
sheets.google.com에서 새 스프레드시트 생성.
1행 헤더: 타임스탬프 사번 성명 출근 퇴근 체류(분) 기기ID 행사ID 결과
2
Apps Script 열기 및 코드 입력
확장 프로그램 → Apps Script → 기존 코드 삭제 후 아래 코드 붙여넣기
function doPost(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getActiveSheet();
    var data = JSON.parse(e.postData.contents);

    // 중복 인증 체크 (같은 행사ID + 기기ID)
    if (data.action === 'checkDuplicate') {
      var rows = sheet.getDataRange().getValues();
      for (var i = 1; i < rows.length; i++) {
        if (rows[i][7] === data.eventId &&
            rows[i][6] === data.deviceId) {
          return ContentService
            .createTextOutput(JSON.stringify({
              status: 'duplicate',
              empName: rows[i][2],
              checkIn: rows[i][3]
            }))
            .setMimeType(ContentService.MimeType.JSON);
        }
      }
      return ContentService
        .createTextOutput(JSON.stringify({status: 'ok'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // 인증 기록 저장
    if (data.action === 'save') {
      sheet.appendRow([
        new Date().toLocaleString('ko-KR'),
        data.empId,
        data.empName,
        data.checkIn,
        data.checkOut,
        data.elapsed,
        data.deviceId,
        data.eventId,
        data.result
      ]);
      return ContentService
        .createTextOutput(JSON.stringify({status: 'ok'}))
        .setMimeType(ContentService.MimeType.JSON);
    }
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({
        status: 'error', message: err.toString()
      }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status: 'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
